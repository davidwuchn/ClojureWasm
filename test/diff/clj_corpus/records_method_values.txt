(every? Character/isWhitespace "   ")
;;=> true
(every? Character/isWhitespace " a ")
;;=> false
(map Character/toUpperCase "ab")
;;=> (\A \B)
(map Character/isDigit "a1")
;;=> (false true)
((juxt Math/abs) -5)
;;=> [5]
(Math/abs -5)
;;=> 5
(sort-by Math/abs [-3 1 -2])
;;=> (1 -2 -3)
(filter Character/isLetter "a1b2")
;;=> (\a \b)
(do (defrecord Pt [x y]) (pr-str (->Pt 1 2)))
;;=> "#user.Pt{:x 1, :y 2}"
(do (defrecord Pt [x y]) (hash (->Pt 1 2)))
;;=> -1763326040
(do (defrecord Pt [x y]) (pr-str (assoc (->Pt 1 2) :z 9)))
;;=> "#user.Pt{:x 1, :y 2, :z 9}"
(do (defrecord Pt [x y]) (= (->Pt 1 2) (read-string "#user.Pt{:x 1, :y 2}")))
;;=> true
(do (defrecord Pt [x y]) (:x (read-string "#user.Pt{:x 7, :y 2}")))
;;=> 7
(do (defrecord Pt [x y]) (hash (assoc (->Pt 1 2) :z 9)))
;;=> -1759466269

;;=> ("1" "true" "a")

(map Math/floor [1.7 2.3])
;;=> (1.0 2.0)
