(subvec [1 2 3 4 5] 1 3)
;;=> [2 3]
(subvec [1 2 3 4 5] 2)
;;=> [3 4 5]
(mapv inc [1 2 3])
;;=> [2 3 4]
(filterv even? [1 2 3 4])
;;=> [2 4]
(into [1 2] [3 4])
;;=> [1 2 3 4]
(into [] (map inc) [1 2 3])
;;=> [2 3 4]
(into #{} [1 1 2])
;;=> #{1 2}
(into {} [[:a 1] [:b 2]])
;;=> {:a 1, :b 2}
(assoc [1 2 3] 1 :x)
;;=> [1 :x 3]
(assoc [1 2 3] 3 :appended)
;;=> [1 2 3 :appended]
(update [1 2 3] 0 inc)
;;=> [2 2 3]
(update [10 20] 1 + 5)
;;=> [10 25]
(get [1 2 3] 5 :default)
;;=> :default
(nth [1 2 3] 5 :default)
;;=> :default
(peek [1 2 3])
;;=> 3
(pop [1 2 3])
;;=> [1 2]
(assoc-in {:a {:b 1}} [:a :c] 9)
;;=> {:a {:b 1, :c 9}}
(update-in {:a {:b 1}} [:a :b] + 10)
;;=> {:a {:b 11}}
(get-in {:a {:b 1}} [:a :x] :nf)
;;=> :nf
(replace {1 :one 2 :two} [1 2 3])
;;=> [:one :two 3]
(mapv vector [1 2] [3 4])
;;=> [[1 3] [2 4]]
(reduce-kv (fn [m k v] (assoc m k (inc v))) {} {:a 1 :b 2})
;;=> {:a 2, :b 3}
(into (sorted-map) {:b 2 :a 1})
;;=> {:a 1, :b 2}
(subvec [1 2 3] 0 0)
;;=> []
