(let [[a b] [1 2]] [a b])
;;=> [1 2]
(let [[a b & r] [1 2 3 4]] r)
;;=> (3 4)
(let [[a & r :as all] [1 2 3]] [a r all])
;;=> [1 (2 3) [1 2 3]]
(let [[a b c] [1]] [a b c])
;;=> [1 nil nil]
(let [{:keys [a b]} {:a 1 :b 2}] [a b])
;;=> [1 2]
(let [{:keys [a b] :or {b 9}} {:a 1}] [a b])
;;=> [1 9]
(let [{:keys [a] :as m} {:a 1 :b 2}] [a m])
;;=> [1 {:a 1, :b 2}]
(let [{a :a b :b} {:a 1 :b 2}] [a b])
;;=> [1 2]
(let [{:strs [a b]} {"a" 1 "b" 2}] [a b])
;;=> [1 2]
(let [{:syms [a b]} {(quote a) 1 (quote b) 2}] [a b])
;;=> [1 2]
(let [[[a b] [c d]] [[1 2] [3 4]]] [a b c d])
;;=> [1 2 3 4]
(let [{[x y] :pt} {:pt [3 4]}] [x y])
;;=> [3 4]
(let [{:keys [a b c] :or {c 0}} {:a 1 :b 2}] [a b c])
;;=> [1 2 0]
((fn [[a b]] (+ a b)) [3 4])
;;=> 7
((fn [{:keys [x y]}] (+ x y)) {:x 1 :y 2})
;;=> 3
(let [[a b & r] [1]] [a b r])
;;=> [1 nil nil]
(let [{:keys [a]} nil] a)
;;=> nil
(let [[a] nil] a)
;;=> nil
(loop [[h & t] [1 2 3] acc 0] (if h (recur t (+ acc h)) acc))
;;=> 6
(let [{:keys [a] :or {a 5}} {}] a)
;;=> 5
(let [[a b & rest] [1 2 3 4]] [a b rest])
;;=> [1 2 (3 4)]
(let [[a b :as all] [1 2]] [a b all])
;;=> [1 2 [1 2]]
(let [{:keys [x y]} {:x 1 :y 2}] [x y])
;;=> [1 2]
(let [{:keys [x] :or {x 9}} {}] x)
;;=> 9
(let [{a :a b :b} {:a 1 :b 2}] [a b])
;;=> [1 2]
(let [{:strs [a b]} {"a" 1 "b" 2}] [a b])
;;=> [1 2]
(let [{:syms [a]} {'a 5}] a)
;;=> 5
(let [[[a b] [c d]] [[1 2] [3 4]]] [a b c d])
;;=> [1 2 3 4]
(let [{:keys [x] :as m} {:x 1 :y 2}] [x m])
;;=> [1 {:x 1, :y 2}]
((fn [& {:keys [a b]}] [a b]) :a 1 :b 2)
;;=> [1 2]
(let [{:keys [a/b]} {:a/b 7}] b)
;;=> 7
(let [{:keys [a/b c/d]} {:a/b 1 :c/d 2}] [b d])
;;=> [1 2]
(let [{:keys [a/b] :or {b 9}} {}] b)
;;=> 9
(let [{:syms [m/n]} {'m/n 5}] n)
;;=> 5
(let [{:keys [a/b x]} {:a/b 1 :x 2}] [b x])
;;=> [1 2]
((fn [a [b c]] [a b c]) 1 [2 3])
;;=> [1 2 3]
(apply (fn [a b & r] [a b r]) [1 2 3 4])
;;=> [1 2 (3 4)]
