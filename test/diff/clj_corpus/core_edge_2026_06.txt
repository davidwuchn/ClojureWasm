;; --- seq / collection edge ---
(partition 2 1 [1 2 3 4])
;;=> ((1 2) (2 3) (3 4))
(partition-by odd? [1 1 2 2 3])
;;=> ((1 1) (2 2) (3))
(reductions + [1 2 3 4])
;;=> (1 3 6 10)
(keep-indexed (fn [i x] (when (odd? i) x)) [:a :b :c :d])

;;=> {:a 3, :b 3}
