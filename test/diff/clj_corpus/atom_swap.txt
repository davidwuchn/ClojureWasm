(deref (atom 5))
;;=> 5
(let [a (atom 0)] (swap! a inc) (deref a))
;;=> 1
(let [a (atom 0)] (swap! a + 5) (deref a))
;;=> 5
(let [a (atom 0)] (reset! a 99) (deref a))
;;=> 99
(let [a (atom 10)] (compare-and-set! a 10 20))
;;=> true
(let [a (atom 10)] (compare-and-set! a 99 20))
;;=> false
(let [a (atom 0)] (swap-vals! a inc))
;;=> [0 1]
(let [a (atom 5)] (reset-vals! a 10))
;;=> [5 10]
(let [a (atom {:n 0})] (swap! a update :n inc) (deref a))
;;=> {:n 1}
(let [a (atom 0)] (swap! a + 1 2 3) (deref a))
;;=> 6
(let [a (atom [])] (swap! a conj 1) (swap! a conj 2) (deref a))
;;=> [1 2]
(let [a (atom 0)] (dotimes [_ 5] (swap! a inc)) (deref a))
;;=> 5
(let [a (atom 1)] (reset! a 2) (swap! a inc) (deref a))
