(map-indexed vector [10 20 30])
;;=> ([0 10] [1 20] [2 30])
(map-indexed vector [])
;;=> ()
(map-indexed vector (map inc [0 1 2]))
;;=> ([0 1] [1 2] [2 3])
(map-indexed (fn [i x] (* i x)) [5 6 7])
;;=> (0 6 14)
(map-indexed vector "abc")
;;=> ([0 \a] [1 \b] [2 \c])
(map-indexed list (range 5))
;;=> ((0 0) (1 1) (2 2) (3 3) (4 4))
(keep-indexed (fn [i x] (if (odd? i) x)) [10 20 30 40])
;;=> (20 40)
(keep-indexed (fn [i x] x) [])
;;=> ()
(keep-indexed (fn [i x] (if (even? i) [i x])) (map inc (range 6)))
;;=> ([0 1] [2 3] [4 5])
(into [] (map-indexed vector) [:a :b :c])
;;=> [[0 :a] [1 :b] [2 :c]]
(map-indexed vector (filter even? (range 10)))
;;=> ([0 0] [1 2] [2 4] [3 6] [4 8])
(keep-indexed (fn [i x] (when (pos? x) (+ i x))) [-1 5 -2 8])
;;=> (6 11)
