(take 3 (distinct (map (fn [x] (mod x 5)) (range))))
;;=> (0 1 2)
(take 3 (dedupe (map (fn [x] (quot x 3)) (range))))
;;=> (0 1 2)
(take 5 (distinct (map (fn [x] (mod x 7)) (range))))
;;=> (0 1 2 3 4)
(distinct [1 1 2 3 3 3 1 4])
;;=> (1 2 3 4)
(dedupe [1 1 2 3 3 3 1 4])
;;=> (1 2 3 1 4)
(distinct [])
;;=> ()
(dedupe [])
;;=> ()
(distinct [nil nil 1 nil])
;;=> (nil 1)
(dedupe [nil nil 1 nil nil])
;;=> (nil 1 nil)
(distinct ["a" "a" "b"])
;;=> ("a" "b")
(distinct '(1 2 1 3 2))
;;=> (1 2 3)
(dedupe "aabbccaa")
;;=> (\a \b \c \a)
(count (dedupe (concat (repeat 3000 1) [2])))
;;=> 2
