(String/join "," ["a" "b" "c"])
;;=> "a,b,c"
(String/join "," [])
;;=> ""
(String/join "," ["a"])
;;=> "a"
(String/join "-" (list "p" "q"))
;;=> "p-q"
(String/join "," (map str [1 2 3]))
;;=> "1,2,3"
(String/join "," #{"z"})
;;=> "z"
(String/join ", " ["x" "y" "z"])
;;=> "x, y, z"
(String/valueOf (char-array [\a \b \c]))
;;=> "abc"
(String/valueOf (char-array [\a \b \c]) 1 2)
;;=> "bc"
(String/copyValueOf (char-array [\x \y]))
;;=> "xy"
(String/copyValueOf (char-array [\p \q \r]) 0 2)
;;=> "pq"
(String. (char-array [\a \b]))
;;=> "ab"
