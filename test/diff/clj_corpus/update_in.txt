;; update-in (O-033 in-Zig -update-in) — clj-parity goldens (values verified vs
;; clojure -M -e 2026-06-15; map print-order is AD-001). The .clj update-in routes
;; a non-empty VECTOR path to -update-in (Zig descent/ascent); non-vector/empty
;; paths fall back to the .clj -update-in-idx (list-path golden below covers it).
(update-in {:a {:b {:c 0}}} [:a :b :c] inc)
;;=> {:a {:b {:c 1}}}
(update-in {:a 1} [:a] + 10)
;;=> {:a 11}
(update-in {} [:a :b] (fnil inc 0))
;;=> {:a {:b 1}}
(update-in [0 [1 2]] [1 0] inc)
;;=> [0 [2 2]]
(update-in {:a {:b 5}} [:a :b] dec)
;;=> {:a {:b 4}}
(update-in {:a {:b {:c {:d 1}}}} [:a :b :c :d] + 9)
;;=> {:a {:b {:c {:d 10}}}}
(update-in {:counts {:a 1}} [:counts :a] inc)
;;=> {:counts {:a 2}}
(get-in (update-in {:m {:n 0}} [:m :n] inc) [:m :n])
;;=> 1
(update-in {:a {:b 1}} '(:a :b) inc)
;;=> {:a {:b 2}}
(update-in [10 20 30] [1] inc)
;;=> [10 21 30]
