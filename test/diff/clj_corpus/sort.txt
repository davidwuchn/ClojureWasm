(sort [])
;;=> ()
(sort [5])
;;=> (5)
(sort [3 1 2])
;;=> (1 2 3)
(sort [3 1 2 1 3 2])
;;=> (1 1 2 2 3 3)
(sort [-5 0 5 -10 10])
;;=> (-10 -5 0 5 10)
(sort [1 1.0 1 2.0 2])
;;=> (1 1.0 1 2.0 2)
(sort ["banana" "apple" "cherry" "apple"])
;;=> ("apple" "apple" "banana" "cherry")
(sort [:c :a :b])
;;=> (:a :b :c)
(sort '(3 1 2))
;;=> (1 2 3)
(sort [[1 2] [1 1] [0 9]])
;;=> ([0 9] [1 1] [1 2])
(sort > [1 3 2 5 4])
;;=> (5 4 3 2 1)
(sort < [5 3 4 1 2])
;;=> (1 2 3 4 5)
(sort-by - [3 1 2])
;;=> (3 2 1)
(sort-by first [[1 :a] [1 :b] [0 :c]])
;;=> ([0 :c] [1 :a] [1 :b])
(sort-by count ["aaa" "a" "aa"])
;;=> ("a" "aa" "aaa")
(take 3 (sort [9 8 7 6 5 4 3 2 1 0]))
;;=> (0 1 2)
(sort [100 2 33 4])
;;=> (2 4 33 100)
