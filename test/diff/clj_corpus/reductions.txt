(reductions + [1 2 3 4 5])
;;=> (1 3 6 10 15)
(reductions + 100 [1 2 3])
;;=> (100 101 103 106)
(reductions + [])
;;=> (0)
(reductions + 5 [])
;;=> (5)
(reductions * [1 2 3 4])
;;=> (1 2 6 24)
(reductions conj [] [1 2 3])
;;=> ([] [1] [1 2] [1 2 3])
(take 5 (reductions + (range)))
;;=> (0 1 3 6 10)
(reductions + [42])
;;=> (42)
(reductions max [3 1 4 1 5 9 2 6])
;;=> (3 3 4 4 5 9 9 9)
(reductions (fn [a x] (if (> a 5) (reduced a) (+ a x))) 0 [1 2 3 4 5 6])
;;=> (0 1 3 6 6)
(into [] (reductions + (range 10)))
;;=> [0 1 3 6 10 15 21 28 36 45]
(reductions str "" ["a" "b" "c"])
;;=> ("" "a" "ab" "abc")
(last (reductions + (range 1000)))
;;=> 499500
(reductions - 10 [1 2 3])
;;=> (10 9 7 4)
