; clojure.pprint/cl-format ~V / ~# runtime-valued directive params (D-458).
; V pulls + CONSUMES the next operand as the param value; # is the count of
; args remaining to process (no consume). ~n[ / ~#[ select a clause by a prefix
; param, consuming no arg. Pinned vs the clj oracle (single-V cases are
; clj-consistent; consecutive-V-on-float is a clj internal-inconsistency bug,
; not pinned — see D-458 discharge).
(clojure.pprint/cl-format nil "~VD" 5 42)
;;=> "   42"
(clojure.pprint/cl-format nil "~V,'*D" 5 42)
;;=> "***42"
(clojure.pprint/cl-format nil "~#D" 42 0 0)
;;=> " 42"
(clojure.pprint/cl-format nil "~VR" 16 255)
;;=> "ff"
(clojure.pprint/cl-format nil "~,V$" 3 12.5)
;;=> "012.50"
(clojure.pprint/cl-format nil "~8,VF" 2 3.14159)
;;=> "    3.14"
(clojure.pprint/cl-format nil "~V,2F" 8 3.14159)
;;=> "    3.14"
(clojure.pprint/cl-format nil "~#[a~;b~;c~]" :x :y)
;;=> "c"
(clojure.pprint/cl-format nil "~#[a~;b~;c~]" :x)
;;=> "b"
(clojure.pprint/cl-format nil "~1[a~;b~;c~]")
;;=> "b"
; D-465: ~F without a d-param — natural (shortest round-trip) value, plain fixed
; notation (never scientific), always with a decimal point. clj-oracle pinned.
(clojure.pprint/cl-format nil "~F" 3.14159)
;;=> "3.14159"
(clojure.pprint/cl-format nil "~F" 3.0)
;;=> "3.0"
(clojure.pprint/cl-format nil "~F" 42)
;;=> "42.0"
(clojure.pprint/cl-format nil "~8F" 3.14159)
;;=> " 3.14159"
(clojure.pprint/cl-format nil "~F" 1.0e10)
;;=> "10000000000.0"
(clojure.pprint/cl-format nil "~F" 0.0001)
;;=> "0.0001"
(clojure.pprint/cl-format nil "~F" 1.5e-7)
;;=> "0.00000015"
(clojure.pprint/cl-format nil "~F" -1.0e10)
;;=> "-10000000000.0"
; ~mincol,colinc,minpad,padchar A/S column padding + ~mincol,padchar,commachar,
; comma-interval :d custom group char (2026-06-23 sweep — were silently ignored).
(clojure.pprint/cl-format nil "~10a|" "hi")
;;=> "hi        |"
(clojure.pprint/cl-format nil "~10@a|" "hi")
;;=> "        hi|"
(clojure.pprint/cl-format nil "~5a|" "longer")
;;=> "longer|"
(clojure.pprint/cl-format nil "~5,,2a|" "ab")
;;=> "ab   |"
(clojure.pprint/cl-format nil "~,,'.,3:d" 1234567)
;;=> "1.234.567"
(clojure.pprint/cl-format nil "~,,'.:d" 1234567)
;;=> "1.234.567"
