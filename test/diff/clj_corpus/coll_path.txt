(assoc-in {:a {:b 1}} [:a :c] 2)
;;=> {:a {:b 1, :c 2}}
(update-in {:a {:b 1}} [:a :b] inc)
;;=> {:a {:b 2}}
(get-in {:a {:b {:c 42}}} [:a :b :c])
;;=> 42
(get-in {:a 1} [:x :y] :default)
;;=> :default
(update-in {:a 0} [:a] + 5)
;;=> {:a 5}
(assoc-in {} [:a :b :c] 1)
;;=> {:a {:b {:c 1}}}
(merge {:a 1} {:b 2} {:a 3})
;;=> {:a 3, :b 2}
(merge-with + {:a 1} {:a 2 :b 3})
;;=> {:a 3, :b 3}
(select-keys {:a 1 :b 2 :c 3} [:a :c])
;;=> {:a 1, :c 3}
(dissoc {:a 1 :b 2 :c 3} :a :c)
;;=> {:b 2}
(update {:a 1} :a inc)
;;=> {:a 2}
(update {} :a (fnil inc 0))
;;=> {:a 1}
(merge nil {:a 1})
;;=> {:a 1}
(into {} (map (fn [[k v]] [k (* v 2)]) {:a 1 :b 2}))
;;=> {:a 2, :b 4}
(reduce-kv assoc {} {:a 1 :b 2})
;;=> {:a 1, :b 2}
(get-in [[1 2] [3 4]] [1 0])
;;=> 3
(update-in [1 2 3] [1] inc)
;;=> [1 3 3]
(assoc [1 2 3] 1 99)
;;=> [1 99 3]
(zipmap [:a :b] [1 2])
;;=> {:a 1, :b 2}
(frequencies "aabbbc")
;;=> {\a 2, \b 3, \c 1}
