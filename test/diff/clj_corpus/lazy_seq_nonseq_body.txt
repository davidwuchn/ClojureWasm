(lazy-seq [2 3])
;;=> (2 3)
(first (lazy-seq [2 3]))
;;=> 2
(rest (lazy-seq [2 3]))
;;=> (3)
(seq (cons 1 (lazy-seq [2 3])))
;;=> (1 2 3)
(vec (cons 1 (lazy-seq [2 3])))
;;=> [1 2 3]
(= (cons 1 (lazy-seq [2 3])) (list 1 2 3))
;;=> true
(take 3 (lazy-seq [1 2 3 4]))
;;=> (1 2 3)
(count (lazy-seq [5 6 7]))
;;=> 3
(lazy-seq (vec (range 3)))
;;=> (0 1 2)
(map inc (lazy-seq [10 20]))
;;=> (11 21)
(reduce + (lazy-seq [1 2 3 4]))
;;=> 10
(into [] (lazy-seq [9 8 7]))
;;=> [9 8 7]
(first (lazy-seq (sorted-set 3 1 2)))
;;=> 1
