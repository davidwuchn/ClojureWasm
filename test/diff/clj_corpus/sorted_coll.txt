(clojure.set/difference #{1 2 3} #{2})
;;=> #{1 3}
(clojure.set/subset? #{1} #{1 2})
;;=> true
(clojure.set/superset? #{1 2} #{1})
;;=> true
(clojure.set/rename-keys {:a 1 :b 2} {:a :x})
;;=> {:b 2, :x 1}
(clojure.set/map-invert {:a 1 :b 2})
;;=> {1 :a, 2 :b}
(sort (sorted-set 3 1 2))
;;=> (1 2 3)
(first (sorted-map :b 2 :a 1))
;;=> [:a 1]
(subseq (sorted-set 1 2 3 4 5) > 2)
;;=> (3 4 5)
(rsubseq (sorted-set 1 2 3) < 3)
;;=> (2 1)
(into (sorted-map) {:c 3 :a 1 :b 2})
;;=> {:a 1, :b 2, :c 3}
(get (sorted-map :a 1) :a)
;;=> 1
(keys (sorted-map :b 2 :a 1))
;;=> (:a :b)
(disj (sorted-set 1 2 3) 2)
;;=> #{1 3}
(conj (sorted-set 1 3) 2)
;;=> #{1 2 3}
(contains? (sorted-set 1 2) 1)
;;=> true
(nth (sort [3 1 2]) 0)
;;=> 1
(rseq [1 2 3])
;;=> (3 2 1)
